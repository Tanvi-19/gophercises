package cmd

import (
        "fmt"
        "strconv"

        db "github.com/gophercises/activity/database"
        "github.com/spf13/cobra"
)

var doCmd = &cobra.Command{
        Use:   "complete",
        Short: "Complete is use to mark task as done",
        Run: func(cmd *cobra.Command, args []string) {
                var nums []int
                for _, num := range args{
                        result, err := strconv.Atoi(num)
                        if err != nil{
                                fmt.Println("Some error occured while parsing ",num)
                        }else{
                                nums = append(nums, result)
                        }
                }
                activities, err := db.ViewAllTasks()
                if err != nil {
                        fmt.Println("Some error occurred ",err)
                        return
                }
                for _, num := range nums{
                        if num <= 0 || num > len(activities){
                                fmt.Println("Wrong task number",num)
                                continue
                        }
                        activity := activities[num-1]
                        err := db.DelTask(activity.Num)
                        if err != nil{
                                fmt.Printf("\"%s\" enable to done",activity.Task)
                        }else{
                                fmt.Printf("Successfully completed task- \"%s\" ",activity.Task)
                        }
                }
                fmt.Println(nums)
        },
}

func init() {
        Rcmd.AddCommand(doCmd)

}

package cmd

import (
        "fmt"
        "os"
        db "github.com/gophercises/activity/database"
        "github.com/spf13/cobra"
)

var listCmd = &cobra.Command{
        Use:   "list",
        Short: "All tasks are here",
        Run: func(cmd *cobra.Command, args []string) {
                activities, err := db.ViewAllTasks()
                if err != nil{
                        fmt.Println("Error occurred :-",err.Error())
                        os.Exit(1)
                }
                if len(activities) == 0{
                        fmt.Println("There is no task to complete")
                        return
                }
                fmt.Println("List of the tasks as follows:-")
                for num, task := range activities{
                        fmt.Printf("%d. %s \n",num+1, task.Task)
                }
        },
}

func init() {
        Rcmd.AddCommand(listCmd)
}
